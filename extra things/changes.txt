def select_dropdown(label, option):

    # Click dropdown using its label
    dropdown = wait.until(EC.element_to_be_clickable((
            By.XPATH,
            f"//label[normalize-space()='{label}']/ancestor::div[contains(@class,'v-input')]//div[@role='button']"
        )))

    dropdown.click()
    wait.until(
    EC.visibility_of_element_located((By.XPATH, "//div[contains(@class,'menuable__content__active')]")))

    time.sleep(1)

    # Wait for dropdown menu
    wait.until(
        EC.visibility_of_element_located(
            (By.XPATH, "//div[@role='listbox']")
        ) )

    while True:

        try:

            item = driver.find_element(
                By.XPATH,
                f"//div[@role='option']//div[normalize-space()='{option}']"
            )

            driver.execute_script(
                "arguments[0].scrollIntoView({block:'center'});",
                item
            )

            driver.execute_script("arguments[0].click();", item)

            break

        except NoSuchElementException:

            listbox = driver.find_element(
                By.XPATH,
                "//div[@role='listbox']"
            )

            driver.execute_script(
                "arguments[0].scrollTop += 250;",
                listbox
            )

            time.sleep(.3)

def open_dropdown(dropdown_id, option):

        EC.element_to_be_clickable((By.XPATH,f"//input[@id='{dropdown_id}']/ancestor::div[@role='button']")))

    driver.execute_script("arguments[0].scrollIntoView({block:'center'});", dropdown)
    dropdown.click()

    option = wait.until(EC.element_to_be_clickable((By.XPATH,"//div[@role='option'][.//div[normalize-space()='INDIA']]")))
    option.click()
